安装mysql
	docker pull mysql
	
	Master(主)
		docker run -p 33061:3306 --name master -e MYSQL_ROOT_PASSWORD=123456 -d mysql:latest
		docker exec -it master /bin/bash
		apt-get update
		apt-get install vim
		
		配置/etc/mysql/my.cnf
		[mysql]
		server-id=100
		#开启二进制日志
		log-bin=mysql-bin
	Slave(从)
		docker run -p 33061:3306 --name slave -e MYSQL_ROOT_PASSWORD=123456 -d mysql:latest
		docker exec -it master /bin/bash
		apt-get update
		apt-get install vim
		
		配置/etc/mysql/my.cnf
		[mysql]
		server-id=101
		#开启二进制日志
		log-bin=mysql-slave-bin
		#中继日志
		relay_log=edu-mysql-real-bin
		
	配好后重启 两个container 
	(主)
	-- 创建账号
	-- create user 'backup'@'%' identified by '123456'
	-- 添加权限
	-- grant REPLICATION SLAVE on *.* to 'backup'@'%'
	-- 查看权限
	-- show grants for 'backup'@'%'
	-- 显示状态
	-- show master status
	(从)
	-- change master to master_host='172.17.0.4', 
		master_user='backup', 
		master_password='123456',
		master_port=3306,
		master_log_file='mysql-bin.000001', 
		master_log_pos=683, 
		master_connect_retry=3;
	-- 显示状态
	-- show slave status
	-- 开启连接
	-- start slave
	-- 停止连接
	-- stop slave
	
	完成！
